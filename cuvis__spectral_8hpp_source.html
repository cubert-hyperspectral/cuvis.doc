134  spectrum_t get_spectrum_polygon(image_t<data_t> const& img, polygon_t const& poly)
135  {
136  //checks if image is reasonable.
137  assert(img._width > 1);
138  assert(img._height > 1);
139  assert(img._channels > 0);
140  assert(img._wavelength != nullptr);
141 
142  // initializing the result
143  spectrum_t res(img._channels);
144 
145  // conversion of polygon relative coordinates to absolute pixel coordinates
146  if (poly.size() > 1) //polygon case
147  {
148 
149  std::vector<cv::Point> transformed_pt; //needs to be integers
150  for (auto const& pt : poly)
151  {
152  transformed_pt.emplace_back(pt._x * (img._width - 1), pt._y * (img._height - 1));
153  }
154 
155  //empty mask
156  cv::Mat mask = cv::Mat::zeros(cv::Size(img._width, img._height), CV_8UC1);
157 
158  // https://docs.opencv.org/4.2.0/db/d75/samples_2cpp_2create_mask_8cpp-example.html#a16
159  // apparently this now has to be wrapped
160 
161  std::vector<std::vector<cv::Point>> vpts;
162  vpts.push_back(transformed_pt);
163 
164  //binary mask
165  //using nearest neighbor
166  cv::fillPoly(mask, vpts, cv::Scalar(255));
167 
168  std::uint64_t n = 0; //counter variable
169  std::vector<std::double_t> sum_v(img._channels, 0.0); //growing sum
170  std::vector<std::double_t> sq_sum_v(img._channels, 0.0); //growing square sum
171 
172  //checking all pixels in polygon mask
173  for (int x = 0; x < img._width; x++)
174  {
175  for (int y = 0; y < img._height; y++)
176  {
177  if (mask.at<std::uint8_t>(y, x) > 128)
178  {
179  n++;
180  for (int z = 0; z < img._channels; z++)
181  {
182  //reading out value
183  auto loc_val = img._data[((y) * img._width + (x)) * img._channels + (z)];
184  //adding up values
185  sum_v[z] += loc_val;
186  sq_sum_v[z] += loc_val * loc_val;
187  }
188  }
189  }
190  }
191 
192  for (int z = 0; z < img._channels; z++)
193  {
194  // convert sum to mean
195  double mean = sum_v[z] / n;
196  spectral_mean_t loc_res;
197  loc_res.value = mean;
198 
199  // sum(x - y)^2 = sum( x^2 - 2xy + y^2) // second binomial formula
200  // Therefore: (sum(x^2) - 2 * sum(x) * y + n*y^2 ) / n
201  // Therefore (sum(x^2) - 2 * sum(x) * y ) / n + y^2
202  loc_res.std = sqrt((sq_sum_v[z] - 2 * sum_v[z] * mean) / n + mean * mean);
203 
204  // set wavelength
205  std::uint32_t loc_wl = img._wavelength[z];
206  loc_res.wavelength = loc_wl;
207  res[z] = loc_res;
208  }
209 
210  return res;
211  }
212  else if (poly.size() == 1) //single point case
213  {
214  //check if out of range
215  if (poly.front()._x > 1.0 || poly.front()._x < 0.0 || poly.front()._y > 1.0 || poly.front()._y < 0.0)
216  {
217  // outside range. nothing to return
218  // TODO throw error?
219  return res;
220  }
221 
222  for (int z = 0; z < img._channels; z++)
223  {
224  //calculate position in memory
225  int y_shift = std::round(poly.front()._y * (img._height - 1)) * img._width;
226  int xy_shift = y_shift + std::round(poly.front()._x * (img._width - 1));
227  //read out point value
228  double loc_val = img._data[xy_shift * img._channels + z];
229  //set wavelength and value
230  std::uint32_t loc_wl = img._wavelength[z];
231  spectral_mean_t loc_res;
232  loc_res.value = loc_val;
233  loc_res.wavelength = loc_wl;
234  res[z] = (loc_res);
235  }
236  return res;
237  }
238  else // poly is empty should never happen
239  {
240  // TODO throw error?
241  return res;
242  }

266  // Yes, the 'const' needs to be removed, OpenCV does not have a cv::ConstMat

268  // ! Make sure that data in this Mat is not modified in this function !
269  const cv::Mat const_img_mat(cv::Size(img._width, img._height), mat_datatype, const_cast<void*>(static_cast<const void*>(img._data)));
270 
271  // Find maximum value for all data
272  double max_val = 0;
273  if (detect_max_value)
274  {
275  cv::minMaxLoc(const_img_mat, nullptr, &max_val, nullptr, nullptr);
276  }
277  else
278  {

280  }
281 
282  const double bin_size = max_val / double(count_bins);
283  const double wlbin_size = double(img._channels) / double(wavelength_bins);
284  const int img_channels_per_wlbin = static_cast<const int>(wlbin_size);
285  const int histogram_count = img._channels / img_channels_per_wlbin;
286 
287  // Calculate histograms
288  for (int count = 0; count < histogram_count; count++)
289  {
290  int bins[] = {int(count_bins)};
291  float range[] = {0.0f, float(max_val)};
292  const float* ranges[] = {range};
293  cv::Mat hist;
294 
295  int channels[] = {0};
296  // Accumulate histograms per channel
297  for (int c = 0; c < img_channels_per_wlbin; c++)
298  {
299  channels[0] = c + count * img_channels_per_wlbin;
300  bool accumulate = c != 0;
301  cv::calcHist(&const_img_mat, 1, channels, cv::Mat(), hist, 1, bins, ranges, true, accumulate);
302  }
303

307  histogram.count = std::vector<std::float_t>(hist.total());
308  histogram.wavelength = img._wavelength[count * img_channels_per_wlbin + int(img_channels_per_wlbin / 2)];
309 
310  // Transfer histogram counts & set x-axis labels
311  for (int idx = 0; idx < hist.total(); idx++)
312  {
313  // Note: calcHist always returns counts as floats
314  histogram.occurrence[idx] = static_cast<std::uint64_t>(reinterpret_cast<float*>(hist.data)[idx]);
315  if (proc_mode == Cube_Reflectance)
316  {

318  }
319  else
320  {
321  histogram.count[idx] = static_cast<std::float_t>(idx * bin_size);
322  }
323  }
324

326  }
