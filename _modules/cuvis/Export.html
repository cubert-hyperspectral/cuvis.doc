    def apply(self, mesu: Measurement) -> Measurement:
        if cuvis_il.status_ok != cuvis_il.cuvis_exporter_apply(self._handle,
                                                               mesu._handle):
            raise SDKException()
        mesu.refresh()
        return mesu
